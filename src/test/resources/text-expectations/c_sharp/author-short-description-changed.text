using Vlingo.Xoom.Common.Version;
using Vlingo.Xoom.Lattice.Model;

namespace Io.Vlingo.Xoomapp.Model.Author;

/**
 * See
 * <a href="https://docs.vlingo.io/xoom-lattice/entity-cqrs#commands-domain-events-and-identified-domain-events">
 *   Commands, Domain Events, and Identified Domain Events
 * </a>
 */
public class AuthorShortDescriptionChanged : IdentifiedDomainEvent
{

  public string Id {get; set;}
  public string ShortDescription {get; set;}

  public AuthorShortDescriptionChanged(string id, string shortDescription) : base(SemanticVersion.From("1.0.0").ToValue())
  {
    this.Id = id;
    this.ShortDescription = shortDescription;
  }

  public override string Identity => Id;
}
