using System;
using Io.Vlingo.Xoomapp.Model.Book;
using Vlingo.Xoom.Common.Serialization;
using Vlingo.Xoom.Symbio;

namespace Io.Vlingo.Xoomapp.Infrastructure.Persistence;

/**
 * See
 * <a href="https://docs.vlingo.io/xoom-lattice/entity-cqrs#stateadapter-and-stateadapterprovider">
 *   StateAdapter and StateAdapterProvider
 * </a>
 */
public class BookStateAdapter : StateAdapter<BookState, TextState>
{

  public override int TypeVersion => 1;

  public override BookState FromRawState(TextState raw) => (BookState) JsonSerialization.Deserialized(raw.Data, raw.Typed)!;

  public override ST FromRawState<ST>(TextState raw) => JsonSerialization.Deserialized<ST>(raw.Data);

  public override TextState ToRawState(string id, BookState state, int stateVersion, Metadata metadata) {
    var serialization = JsonSerialization.Serialized(state);
    return new TextState(id, typeof(BookState), TypeVersion, serialization, stateVersion, metadata);
  }
}
